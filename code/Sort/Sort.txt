private static void swap(int i, int j) {
	int tmp = array[i];
	array[i] = array[j];
	array[j] = tmp;
}

@Obfuscate
public void sort() {
	for (int i = 0; i < size; i++) {
		boolean isSwapped = false;
		for (int j = size - 1; j > i; j--) {
			if (array[j] < array[j - 1]) {
				swap(j, j - 1);
				isSwapped = true;
			}
		}
		if (!isSwapped) {
			break;
		}
	}
}
